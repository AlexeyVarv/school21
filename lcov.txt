lcov_report:
	lcov --capture --initial --directory . --output-file a_out_lcov_base.info
	lcov --capture --directory . --output-file a_out_lcov_test.info
	lcov --add-tracefile a_out_lcov_base.info --add-tracefile a_out_lcov_test.info --output-file a_out_lcov_total.info
	genhtml a_out_lcov_total.info --output-directory a_out_lcov_html --demangle-cpp --legend --title "Cpp basic gcov test"
